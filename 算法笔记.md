11.结构体初始化

```C++
struct studentInfo{
		int id;
		char gender;
		//可以不初始化就定义结构体变量
		studentInfo(){}
		//只初始化gender
		studentInfo(char _gender){
			gender = _gender
		}
		//初始化 gender和id
		studentInfo(){
			id = _id;
			gender = _gender;
		}
};
```
